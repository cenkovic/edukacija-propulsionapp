show() & hide()

$('#showButton').on('click', function() {
    $('#showHideCode').show('slow', function(){
        alert('ELEMENT DISPLAYED');
    });
});
$('#hideButton').on('click', function() {
    $('#showHideCode').hide(5000);
});

toggle()

$('#showButton').on('click', function() {
    $('#showHideCode').show('slow', function(){
        alert('ELEMENT DISPLAYED');
    });
});
$('#hideButton').on('click', function() {
    $('#showHideCode').hide(5000);
});

slideDown() & slideUp()

$('#slideDownButton').on('click', function() {
    $('#slideUpDownCode').slideDown(4523);
});
$('#slideUpButton').on('click', function() {
    $('#slideUpDownCode').hide('fast', function(){
        alert('ELEEMENT HIDDEN');
    });
});

slideToggle()

$('#slideToggleButton').on('click', function() {
    $('#slideToggleCode').slideToggle('slow');
});

fadeIn() & fadeOut()

$('#fadeInButton').on('click', function() {
    $('#fadeInOutCode').fadeIn('slow');
});
$('#fadeOutButton').on('click', function() {
    $('#fadeInOutCode').fadeOut(4000);
});

fadeToggle()

$('#fadeToggleButton').on('click', function() {
    $('#fadeToggleCode').fadeToggle(2500, function(){
        alert('FADED');
    });
});

fadeTo()

$('#fadeToButton').on('click', function() {
    $('#fadeToCode').fadeTo(2500, 0.5, function(){
        alert('FADED TO 0.5');
    });
});

Pažnja! Ako prvo fejdujete element do na primer 0.5 i onda pozovete fade out pa fade in na isti element, element će imati 0.5 opacity.

animate()

Ova funkcija vam omogućuje da animirate attribute elementa. Mogu se animirati CSS atributi koji imaju numeričku vrednost.

Animaciju je moguće zaustaviti koriščenjem funkcije stop() pozivajući je nad objektom nad kojim je pokrenuta animacija.(Klikni na animate, sačekaj par sekundi pa klikni na stop animate) 

//animate
$('#animateButton').on('click', function() {
    $(this).animate({'font-size': 100}, 10000, function(){
        alert('ANIMATION FINISHED');
    });
});
//stop animate
$('#stopButton').on('dblclick', function() {
    $(this).stop();
})
